Card {
  cardSpecifiers Array of CardSpecifier
}

// Including 2 or more of the same specifier types into Card will apply OR to them.

CardSpecifier {
  type CardSpecifierType,
  specifier Choice {
    CardType,
    Player,
    Trait,
    ExactName,
    NameContains,
    Level,
    Color,
    CostSpecifier,
    TriggerIcon,
    Power,
    LevelWithMultiplier,
    State,
    Nothing
  }
}

enum CardSpecifierType {
  cardType = 1,
  owner = 2,
  trait = 3,
  exactName = 4,
  nameContains = 5,
  level = 6, 
  levelHigherThanOpp = 7, // level is higher than your opponent's level
  color = 8,
  cost = 9,
  triggerIcon = 10,
  hasMarker = 11,
  power = 12,
  standbyTarget = 13, // with level equal to or less than your level +1
  levelWithMultiplier = 14,
  state = 15, // stand, rested, reversed
  sumOfLevelsLessThanDiffNamedEventsInMemory = 16 // sum of levels less than number of differently named events in memory aka Hina finisher
}

enum CardType {
  char = 1,
  climax = 2,
  event = 3,
  marker = 4
}

enum Color {
  yellow = 1,
  green = 2,
  red = 3,
  blue = 4,
  purple = 5
}

Trait {
  value String
}

ExactName {
  value String
}

NameContains {
  value String
}

Level {
  value Number
}

CostSpecifier {
  value Number
}

Power {
  value Number
}

LevelWithMultiplier {
  value Number,
  multiplier Multiplier
}
