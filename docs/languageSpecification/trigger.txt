Trigger {
  type TriggerType
  trigger Choice {
    ZoneChangeTrigger,
    PhaseTrigger,
    TriggerRevealTrigger,
    OnPlayTrigger,
    OnAttackTrigger,
    StateChangeTrigger,
    OnBeingAttackedTrigger,
    OnDamageCancelTrigger,
    OnDamageTakenCancelTrigger,
    OnPayingCostTrigger,
    OnActAbillityTrigger,
    OtherTrigger,
    Nothing
  }
}

enum TriggerType {
  onZoneChange = 1,
  onPlay = 2,
  onStateChange = 3,
  onAttack = 4,
  onBackupOfThis = 5,
  onTriggerReveal = 6,
  onPhaseEvent = 7,
  onEndOfThisCardsAttack = 8,
  onOppCharPlacedByStandbyTriggerReveal = 9,
  onBeingAttacked = 10,
  onDamageCancel = 11,
  onDamageTakenCancel = 12,
  onPayingCost = 13,
  onActAbillity = 14,
  
  otherTrigger = 255
}

ZoneChangeTrigger {
  target Array of Target,  // targets are ORed
  from Zone,
  to Zone
}

PhaseTrigger {
  state PhaseState,
  phase Phase,
  player Player
}  

TriggerRevealTrigger {
  card Card
}

OnPlayTrigger {
  target Target
}

OnAttackTrigger {
  target Target // who attacks
}

StateChangeTrigger {
  state State,
  target Target
}

OnBeingAttackedTrigger {
  target Target, // who is being attacked
  attackType AttackType
}

OnDamageCancelTrigger {
  damageDealer Target,
  cancelled Bool
}

// when the damage taken by you isn't cancelled...
OnDamageTakenCancelTrigger {
  cancelled Bool
}

OnPayingCostTrigger {
  abilityType AbilityType, // i.e. when paying for [Auto] of target...
  target Target
}

OnActAbillityTrigger {
  player Player
}

OtherTrigger {
  cardCode String
}
