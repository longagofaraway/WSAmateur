Ability {
  type AbilityType,
  ability Choice {
    AutoAbility,
    ContAbility,
    ActAbility,
    EventAbility
  }
}

enum AbilityType {
  cont = 1,
  auto = 2,
  act = 3,
  event = 4 // for abilities on events, triggers icons, global encore etc...
}

enum Keyword {
  encore = 1,
  cxcombo = 2,
  brainstorm = 3,
  backup = 4,
  experience = 5,
  resonance = 6,
  bond = 7,
  replay = 8,
  alarm = 9,
  change = 10,
  assist = 11
}

// for binary encoding
enum AbilityItem {
  endTag = 0,
  cost = 1,
  trigger = 2,
  effect = 3,
  keyword = 4,
  activation = 5
}

AutoAbility {
  activationTimes UInt8,
  keywords Array of Keyword,
  cost Choice {
    Cost,
    Nothing
  },
  trigger Trigger,
  effects Array of Effect
}

ContAbility {
  keywords Array of Keyword,
  effects Array of Effect
}

ActAbility {
  keywords Array of Keyword,
  cost Cost,
  effects Array of Effect
}

EventAbility {
  keywords Array of Keyword,
  effects Array of Effect
}
